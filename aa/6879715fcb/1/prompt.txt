     ```python
     VALID_CATEGORIES: frozenset = frozenset({
         "IDE/Editor", "CLI Tool", "Testing", "Build/Deploy",
         "Monitoring/Observability", "Database", "API/SDK", "DevOps",
         "Code Quality", "Package Manager", "Other",
     })

     def get_devtools_categories_batch(candidates: List[Dict[str, str]]) -> Dict[str, Optional[str]]:
         results: Dict[str, Optional[str]] = {}
         pending: List[Dict[str, str]] = []
         for candidate in candidates:
             cid = candidate["id"]
             name = candidate.get("name", "")
             text = candidate.get("text", "")
             key = _cache_key(name, text)
             cached = _cache_get(_category_cache, key)
             if cached is not None:
                 results[cid] = cached
             else:
                 pending.append({**candidate, "_cache_key": key})
         if not pending:
             return results
         if not _has_openai_key():
             for candidate in pending:
                 results[candidate["id"]] = None
             return results
         chunks = [pending[i:i + _BATCH_SIZE] for i in range(0, len(pending), _BATCH_SIZE)]
         for chunk in chunks:
             payload = [{"item_id": c["id"], "name": c.get("name", ""), "description": c.get("text", "")} for c in chunk]
             result_map: Dict[str, str] = {}
             try:
                 response = _call_openai(
                     [{"role": "system", "content": ('Classify each devtool into exactly one category. '
                         'Valid categories: IDE/Editor, CLI Tool, Testing, Build/Deploy, '
                         'Monitoring/Observability, Database, API/SDK, DevOps, Code Quality, '
                         'Package Manager, Other. '
                         'Respond with JSON: {"results": {"<item_id>": "<category>", ...}}.')},
                      {"role": "user", "content": json.dumps(payload)}],
                     max_tokens=len(payload) * 15 + 50,
                     temperature=0.0,
                     response_format={"type": "json_object"},
                 )
                 content = response.choices[0].message.content.strip()
                 data = json.loads(content)
                 result_map = data.get("results", {})
             except Exception:
                 logger.exception("classifier.category_batch_failure", extra={"event": "classifier.category_batch_failure"})
             for candidate in chunk:
                 cid = candidate["id"]
                 if not result_map:
                     category = get_devtools_category(candidate.get("text", ""), candidate.get("name", ""))
                 else:
                     raw = result_map.get(cid)
                     if raw is None:
                         category = get_devtools_category(candidate.get("text", ""), candidate.get("name", ""))
                     else:
                         category = raw if raw in VALID_CATEGORIES else None
                 if category is not None:
                     _cache_set(_category_cache, candidate["_cache_key"], category)
                 results[cid] = category
         return results
     ```
